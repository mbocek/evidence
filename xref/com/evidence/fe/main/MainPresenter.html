21  import org.slf4j.Logger;
22  import org.slf4j.LoggerFactory;

27  import org.vaadin.mvp.presenter.FactoryPresenter;
28  import org.vaadin.mvp.presenter.IPresenter;
29  import org.vaadin.mvp.presenter.IPresenterFactory;
30  import org.vaadin.mvp.presenter.annotation.Presenter;

32  import com.evidence.fe.EvidenceApplication;
33  import com.evidence.fe.menu.MenuPresenter;
34  import com.vaadin.ui.HorizontalSplitPanel;
35  import com.vaadin.ui.VerticalLayout;
36  import com.vaadin.ui.Window;
37  
38  /**
39   * @author Michal Bocek
40   * @since 1.0.0
41   */
42  @Component("mainPresenter")
43  @Scope("prototype")
44  @Presenter(view = MainView.class)
45  public class MainPresenter extends FactoryPresenter<IMainView, MainEventBus> {
46  
47  	private static final Logger log = LoggerFactory.getLogger(MainPresenter.class); // NOPMD
48  
49  	private EvidenceApplication application;
50  
51  	private MenuPresenter menuPresenter; // NOPMD
52  
53  	private IPresenter<?, ? extends EventBus> contentPresenter; // NOPMD
54  
55  	public void onStart(final EvidenceApplication app) {
56  		// keep a reference to the application instance
57  		this.application = app;
58  
59  		// set the applications main windows (the view)
60  		this.application.setMainWindow((Window) this.view);
61  
62  		//this.application.setTheme("my-chameleon");
63  		
64  		// load the menu presenter
65  		this.menuPresenter = (MenuPresenter) this.application.getPresenterFactory().createPresenter("menuPresenter");
66  		this.view.setMenu(this.menuPresenter.getView());
67  	}
68  
69  	public void onOpenModule(final Class<? extends BasePresenter<?, ? extends EventBus>> presenter) {
70  		// opening module invoked from menu 
71  		log.debug("Openning module for presenter {}", presenter.getCanonicalName());
72  		final IPresenterFactory pf = this.application.getPresenterFactory();
73  		this.contentPresenter = pf.createPresenter(presenter.getAnnotation(Component.class).value());
74  		this.view.setContent((com.vaadin.ui.Component) this.contentPresenter.getView());
75  	}
76  
77  	public void onShowDialog(final Window dialog) {
78  		this.application.getMainWindow().addWindow(dialog);
79  	}
80  
81  	@Override
82  	public void bind() {
83  		final VerticalLayout mainLayout = this.view.getMainLayout();
84  		final HorizontalSplitPanel layoutPanel = this.view.getSplitLayout();
85  		mainLayout.setExpandRatio(layoutPanel, 1.0f);
86  		layoutPanel.setSplitPosition(25, HorizontalSplitPanel.UNITS_PERCENTAGE);
87  	}
88  }
