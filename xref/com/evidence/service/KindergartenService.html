19  package com.evidence.service;
20  
21  import java.util.List;
22  
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.stereotype.Service;
25  import org.springframework.transaction.annotation.Transactional;
26  
27  import com.evidence.dao.KindergartenDAO;
28  import com.evidence.dto.KindergartenDTO;
29  import com.evidence.entity.Kindergarten;
30  import com.evidence.utility.DTOConverter;
31  
32  /**
33   * @author Michal Bocek
34   * @since 1.0.0
35   */
36  @Service
37  @Transactional(readOnly = true)
38  public class KindergartenService {
39  
40  	@Autowired
41  	private KindergartenDAO kindergartenDao;
42  
43  	public List<KindergartenDTO> getAll() {
44  		final List<Kindergarten> kindergartens = kindergartenDao.findAll();
45  		return DTOConverter.convertList(kindergartens, KindergartenDTO.class);
46  	}
47  
48  	public void addKindergarten(final KindergartenDTO kindergartenDTO) {
49  		final Kindergarten kindergarten = DTOConverter.convert(kindergartenDTO, Kindergarten.class);
50  		kindergartenDao.create(kindergarten);
51  	}
52  }
