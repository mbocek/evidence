19  package com.tapas.evidence.utility.converter;
20  
21  import org.dozer.CustomConverter;
22  import org.dozer.MappingException;
23  
24  import com.tapas.evidence.dto.ContactDTO;
25  import com.tapas.evidence.dto.ResponsiblePersonDTO;
26  import com.tapas.evidence.entity.Contact;
27  import com.tapas.evidence.entity.Kindergarten;
28  import com.tapas.evidence.entity.ResponsibilityType;
29  import com.tapas.evidence.entity.ResponsiblePerson;
30  import com.tapas.evidence.utility.DTOConverter;
31  
32  /**
33   * @author Michal Bocek
34   * @since 1.0.0
35   */
36  public class ResponsiblePersonConverter implements CustomConverter {
37  
38  	/* (non-Javadoc)
39  	 * @see org.dozer.CustomConverter#convert(java.lang.Object, java.lang.Object, java.lang.Class, java.lang.Class)
40  	 */
41  	@Override
42  	public Object convert(final Object destination, final Object source, final Class<?> destClass, final Class<?> sourceClass) {
43  		Object result = null;
44  		if (source instanceof ResponsiblePersonDTO) {
45  			final ResponsiblePerson responsiblePerson = convert((ResponsiblePersonDTO) source, (ResponsiblePerson) destination);
46  			result = responsiblePerson;
47  		} else if (source instanceof ResponsiblePerson) {
48  			final ResponsiblePersonDTO responsiblePerson = convert((ResponsiblePerson) source);
49  			result = responsiblePerson;
50  		} else if (source != null) {
51  			throw new MappingException("Converter ResponsiblePersonConverter used incorrectly. Arguments passed were:"
52  					+ destination + " and " + source);
53  		}
54  		return result;
55  	}
56  
57  	private ResponsiblePersonDTO convert(final ResponsiblePerson source) {
58  		final ResponsiblePersonDTO responsiblePersonDTO = new ResponsiblePersonDTO(source.getKindergarten().getId());
59  		responsiblePersonDTO.setContact(DTOConverter.convert(source.getContact(), ContactDTO.class));
60  		responsiblePersonDTO.setBirthDate(source.getBirthDate());
61  		responsiblePersonDTO.setName(source.getName());
62  		responsiblePersonDTO.setSurName(source.getSurName());
63  		responsiblePersonDTO.setType(source.getType().name());
64  		responsiblePersonDTO.setId(source.getId());
65  		responsiblePersonDTO.setPhoto(source.getPhoto());
66  		return responsiblePersonDTO;
67  	}
68  
69  	private ResponsiblePerson convert(final ResponsiblePersonDTO sourceDTO, final ResponsiblePerson destination) {
70  		final ResponsiblePerson responsiblePerson = destination == null ? new ResponsiblePerson() : destination;
71  		Contact contact = responsiblePerson.getContact();
72  		if (contact == null) {
73  			contact = DTOConverter.convert(sourceDTO.getContact(), Contact.class);
74  		} else {
75  			DTOConverter.convert(sourceDTO.getContact(), contact);
76  		}
77  		responsiblePerson.setContact(contact);
78  		responsiblePerson.setBirthDate(sourceDTO.getBirthDate());
79  		responsiblePerson.setKindergarten(new Kindergarten(sourceDTO.getKindergartenId()));
80  		responsiblePerson.setName(sourceDTO.getName());
81  		responsiblePerson.setSurName(sourceDTO.getSurName());
82  		responsiblePerson.setType(ResponsibilityType.valueOf(sourceDTO.getType()));
83  		responsiblePerson.setPhoto(sourceDTO.getPhoto());
84  		return responsiblePerson;
85  	}
